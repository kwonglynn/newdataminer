Use javascript to do simple filtering on a table.

Reference

1. Create an input box and define a onkeyup function for it.
<input class="w3-input w3-border w3-padding" type="text" placeholder="Search for names.." id="myInput" onkeyup="myFunction()">
2. Create the table, note all the table content has to be on the same page, otherwise only the current page can be searched.
<table id="myTable">
  <tr>
    <td>Data</td>
    ...
  </tr>
</table>
3. Create a javascript to handle the filtering on the tabel cells, using the trick of adding/removing a display class.
<script>
function myFunction() {
  var input, filter, table, tr, td, i;
  input = document.getElementById("myInput");
  filter = input.value.toUpperCase();
  table = document.getElementById("myTable");
  tr = table.getElementsByTagName("tr");
  for (i = 0; i < tr.length; i++) {
    td = tr[i].getElementsByTagName("td")[0];
    if (td) {
      txtValue = td.textContent || td.innerText;
      if (txtValue.toUpperCase().indexOf(filter) > -1) {
        tr[i].style.display = "";
      } else {
        tr[i].style.display = "none";
      }
    }
  }
}
</script>
